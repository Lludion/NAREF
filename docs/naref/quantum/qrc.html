 58	def __reset_seq(self) -> None:
 59		""" Sets self.set to an initial, blank sequence, with the right layout, device and channels """

 63	def layout(self) -> Register:
 64		""" Returns the right Register object associated with the lattice of geometry 'self.geom' """
 65
 66		if self.geom == 'grid_lattice':
 67			q_dict = topology.grid_lattice(int(np.sqrt(self.n)), int(np.sqrt(self.n)), self.atom_distance)
 68		elif self.geom == 'grid_lattice_centred':
 69			q_dict = topology.grid_lattice_centred(int(np.sqrt(self.n)), self.atom_distance)
 70		elif self.geom == 'circle_lattice':
 71			q_dict = topology.circle_lattice(self.n, self.atom_distance)
 72		elif self.geom == 'triangle_lattice':
 73			q_dict = topology.triangle_lattice(int(-1 / 2 + np.sqrt(1 / 4 + 2 * self.n)), self.atom_distance)
 74		else:
 75			raise Exception("Unknown Lattice Type")
 76
 77		reg = Register(q_dict)
 78		return reg

 80	def build_reservoir(self, inp_trunc: list, scope: str = "training") -> list:
 81		"""
 82		Encodes input_trunc as input parameters in local pulses and constructs the reservoir from the encoding.
 83		Returns all measurement outcomes associated with the truncated input
 84
 85		The input is here called truncated, as typically, the total dataset is divided into training and test set.
 86		Each can therefore be considered as 'truncated'.
 87
 88		Args:
 89			inp_trunc: truncated input (cf above)
 90			scope: whether to print that we are in the training or testing phase
 91
 92		Returns:
 93			the list [lists of the samples of Z_MEASUREMENT( R(window) ) (for each window)]
 94		"""
 95
 96		self.rese_data = inp_trunc
 97		self.scope = scope
 98
 99		# total time since the last constants.FEEDBACK_INTERVAL steps
100		self.__k_time = 0
101
102		# number of available windows
103		nwin = len(self.rese_data) - self.sample_len
104
105		print(f"Computing {nwin} training steps...")
106
107		# list of outcomes for each window (uK ...uk+tau)
108		return [self.window(k) for k in range(nwin)]
109
110	def window(self, k: int) -> dict:
111		"""
112		For each window (uK ...uK+tau) during the execution,
113		we have to do the sequence of pulses corresponding to R(windowK, rhoK)
114
115		Args:
116			k: index of the window (uK ...uK+tau)
117
118		Returns:
119			the dict of the samples of Z_MEASUREMENT( R(window) )
120		"""
121
122		if self.verbose:
123			tic = time.time()  # measure the time spent in this window
124
125		# initialize Sequence self.seq
126		self.__reset_seq()
127
128		# extract subsection of time series to pass into the reservoir
129		self.wlist = self.rese_data[k:k + self.sample_len]
130
131		# for each ui in (uK ...uK+tau)
132		for ui in range(self.sample_len):
133			self.add_to_sequence(ui)
134
135		sim = QutipEmulator.from_sequence(self.seq, sampling_rate=1)
136		results = sim.run(progress_bar=False)  # run the simulation
137		outcomes = results.sample_final_state(N_samples=self.N_samples)
138
139		if self.verbose:
140			toc = time.time()
141			self.__k_time += (toc - tic)
142			if k % 10 == 9:  # as k starts at 0, this ensures 10 steps have passed before the first feedback
143				print(f"{k + 1} {self.scope} steps finished, {self.scope} time : {self.__k_time:.3f}s")
144				self.__k_time = 0
145
146		return outcomes

148	def add_to_sequence(self, i: int) -> None:
149		"""
150		For each ui in a given window (uK ...uK+tau), we have to add either a corresponding pulse or a reset.
151		Args:
152			i: index of the input under scrutiny
153		"""
154		duration = self.inp_duration  # duration for which the reservoir will see the input = pulse duration (in ns)
155
156		reset = rd.choice(
157			np.arange(0, 2),
158			p=[1 - self.reset_rate, self.reset_rate],
159			size=1)  # in case we want to add random noise
160
161		if reset:
162			print("reset")
163			self.__reset_seq()
164
165		else:
166			amplitude = self.wlist[i]  # input is encoded in the pulse amplitude
167			detuning = 0
168
169			# It seems that Pulser sometimes messes up the random seed at this point.
170			if np.isclose(self.reset_rate, 0):
171				# If we do not need randomness at this point, we reinitialise the seed.
172				rd.seed(constants.DEFAULT_SEED)
173
174			pulse = Pulse.ConstantPulse(duration, amplitude, detuning, phase=0.00, post_phase_shift=0.0)
175			self.seq.add(pulse, "global")

177	def train(self, data: list, train_len: int) -> float:
178		"""
179		Executes the training for all inputs in data.
180		Fills in self.train_data.
181		Constructs the optimised weights from the training.
182		Returns and displays the final total error (train error).
183
184		Args:
185			data: input data
186			train_len: the number of training steps
187		"""
188		self.train_data = data[:self.sample_len + train_len]
189
190		total_time = 0
191		tic = time.time()
192		self.clist = self.build_reservoir(self.train_data)
193
194		# normalise the counts list so that it is coherent between train and test
195		self._fill_nclist(self.clist)
196
197		# The following link justifies using from_dict for a list of dict objects
198		# www.geeksforgeeks.org/create-a-pandas-dataframe-from-list-of-dicts/
199		self.ncdf = pd.DataFrame.from_dict(self.nclist).fillna(0)
200		self.X_all_train = self.ncdf.to_numpy()
201
202		# calculate pseudo inverse
203		xp = np.linalg.pinv(self.X_all_train)
204		# calculate the weight vector (linear regression)
205		self.w = np.dot(xp, self.train_data[self.sample_len:self.sample_len + len(self.clist)])
206
207		toc = time.time()
208		total_time += (toc - tic)
209		print(f"Training finished. Total training time: {total_time:.3f}s")
210
211		self.y_data_train = self.train_data[self.sample_len:self.sample_len + train_len]
212		self.y_pred_train = self._get_prediction(self.X_all_train, len(self.clist))
213		error_train: float = self.get_error_train()
214		print(f"Train error ({constants.ERROR_NAME}): ", error_train)
215		return error_train
216
217	def _get_prediction(self, X_all: np.ndarray, c_len: int) -> list:
218		"""
219		Args:
220			X_all: array version of normalized counts (after pd filling the NaN)
221			c_len: length of the list of counts
222
223		Returns:
224			List of the predictions associated with the reservoir weights w
225		"""
226		return [np.dot(X_all[c, :], self.w) for c in range(c_len)]
227
228	def _fill_nclist(self, counts: list) -> None:
229		"""
230		Normalizes the counts list so that it is coherent between the testing and training phases.
231		Args:
232			counts: list of counts (of sample of window), either from training or testing phase.
233		"""
234		self.nclist = [{k: v / self.N_samples for k, v in c.items()} for c in counts]

259	def evaluate(self, test_data: list, test_len: int) -> float:
260		"""
261		Builds the reservoirs associated with the test data. Returns the observed hidden states and the predictions
262		Args:
263			test_data: input data
264			test_len: the number of testing steps
265		"""
266
267		self.test_data = test_data[:self.sample_len + test_len]
268		self.clist_test = self.build_reservoir(self.test_data, scope="testing")
269
270		# normalise the counts list so that it is coherent between train and test
271		self._fill_nclist(self.clist_test)
272
273		df_copy = self.ncdf.iloc[:0, :].copy().fillna(0)
274		df_copy = df_copy.append(self.nclist, ignore_index=True, sort=False).fillna(0)
275		self.X_all_test = df_copy.to_numpy()
276
277		# evaluate testing error
278		self.y_pred_test = self._get_prediction(self.X_all_test, len(self.clist_test))
279		self.y_data_test = self.test_data[self.sample_len:self.sample_len + test_len]
280
281		error_test: float = self.get_error_test()
282		print(f"Test error ({constants.ERROR_NAME}): ", error_test)
283		return error_test

111	def window(self, k: int) -> dict:
112		"""
113		For each window (uK ...uK+tau) during the execution,
114		we have to do the sequence of pulses corresponding to R(windowK, rhoK)
115
116		Args:
117			k: index of the window (uK ...uK+tau)
118
119		Returns:
120			the dict of the samples of Z_MEASUREMENT( R(window) )
121		"""
122
123		if self.verbose:
124			tic = time.time()  # measure the time spent in this window
125
126		# initialize Sequence self.seq
127		self.__reset_seq()
128
129		# extract subsection of time series to pass into the reservoir
130		self.wlist = self.rese_data[k:k + self.sample_len]
131
132		# for each ui in (uK ...uK+tau)
133		for ui in range(self.sample_len):
134			self.add_to_sequence(ui)
135
136		sim = QutipEmulator.from_sequence(self.seq, sampling_rate=1)
137		results = sim.run(progress_bar=False)  # run the simulation
138		outcomes = results.sample_final_state(N_samples=self.N_samples)
139
140		if self.verbose:
141			toc = time.time()
142			self.__k_time += (toc - tic)
143			if k % 10 == 9:  # as k starts at 0, this ensures 10 steps have passed before the first feedback
144				print(f"{k + 1} {self.scope} steps finished, {self.scope} time : {self.__k_time:.3f}s")
145				self.__k_time = 0
146
147		return outcomes
